set(LSP_SRV ydshd)
add_library(ydshd_static STATIC
        transport.cpp
        server.cpp
        lsp.cpp
        ast.cpp

        RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
        LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR})
